项目结构搭建，路径中不能有中文，否则sqlite找不到文件，而且css加载无效
application()
{
    manage.py               <-脚本数据
    runserver.py            <-启动服务器
    {
        application()       <-web目录
        {
            app.conf        <-配置文件
            models.py       <-数据模型
            views.py        <-视图，url的路径
            __init__.py     <-模块导出文件
        }
        static              <-静态文件
        templates()         <-页面模版
        {
            base.html
            index.html
            login.html
        }
    }
}
